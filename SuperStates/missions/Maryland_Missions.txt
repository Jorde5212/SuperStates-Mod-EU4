			custom_tooltip = myd_religious_lev_effect_tooltip
			tooltip = {
				add_country_modifier = {
					name = myd_religious_complete
					duration = 9125
				}
			}
			custom_tooltip = myd_humanist_lev_effect_tooltip
			tooltip = {
				add_country_modifier = {
					name = myd_humanist_complete
					duration = 9125
				}
			}
			hidden_effect = { #Here is where the actual effects are
				if = {
					limit = {
						OR = {
							religion = catholic
							religion = s_catholic
						}
					}
					add_country_modifier = {
						name = "myd_is_catholic"
						duration = 9125
					}
				}
				if = {
					limit = {
						humanist_ideas = 7
						NOT = {
							religious_ideas = 7
						}
					}
					add_country_modifier = {
						name = "myd_humanist_complete"
						duration = 9125
					}
				}
				if = {
					limit = {
						religious_ideas = 7
						NOT = {
							humanist_ideas = 7
						}
					}
					add_country_modifier = {
						name = "myd_religious_complete"
						duration = 9125
					}
				}
				else = {
					add_adm_power = 200
				}
			}
		}
	}
