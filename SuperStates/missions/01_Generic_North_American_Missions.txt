#New England Regional
gen_new_england = {
	slot = 5
	generic = yes
	ai = yes

	potential = {
		capital_scope = {
			region = new_england_region
		}
	}
	has_country_shield = no

	gen_new_england_inno = {
		icon = mission_rb_renovate_oxbridge
		required_missions = { }
		position = 1
		trigger = {
			university = 15
			all_neighbor_country = {
				NOT = {
					tech_difference = 0
				}
			}
		}
		effect = {
			if = {
				limit = {
					has_dlc = "Rule Britannia"
				}
				add_innovativeness_big_effect = yes
			}
			else = {
				add_adm_power = 50
			}
			add_prestige = 20
		}
	}

	gen_new_england_naval = {
		icon = mission_rb_a_mighty_fleet
		required_missions = { }
		position = 2
		trigger = {
			num_of_light_ship = 30
			navy_size_percentage = 1
		}
		effect = {
			add_country_modifier = {
				name = "new_england_trade_ship"
				duration = 9125 #25 years
			}
		}
	}
}

gen_mid_atlantic = {
	slot = 5
	generic = yes
	ai = yes

	potential = {
		capital_scope = {
			region = mid_atlantic_region
		}
	}
	has_country_shield = no

	gen_mid_atlantic_trade_node = {
		icon = mission_war_chest
		required_missions = { }
		position = 1
		
		trigger = {
			home_trade_node = {
				highest_value_trade_node = yes
				is_strongest_trade_power = ROOT
			}
			
		}
		
		effect = {
			add_prestige = 30
			add_country_modifier = {
				name = "gen_eur_trade_node_mod"
				duration = 7300
			}
		}
	}

	gen_mid_atlantic_banking = {
		icon = medici_bank_ledgers
		required_missions = { }
		position = 2

		trigger = {
			num_of_loans = 0
			treasury = 2000
			custom_trigger_tooltip = {
				tooltip = mid_atlantic_banking_tooltip
				OR = {
					all_neighbor_country = {
						inflation = ROOT
					}
					NOT = { inflation = 0.1 }
				}
			}
		}
		effect = {
			add_country_modifier = {
				name = "mid_atlantic_bankers"
				duration = 9125
			}
		}
	}
}

#Great Plains Regional
gen_greatplains = {
	slot = 5
	generic = yes
	ai = yes

	potential = {
		capital_scope = {
			region = great_plains_region
		}
	}
	has_country_shield = no

	gen_cowboys = {
		icon = mission_cowboy
        required_missions = { }
        position = 1

        provinces_to_highlight = {
            OR = {
                trade_goods = livestock
            }
            NOT = { owned_by = ROOT }
            has_discovered = ROOT
        }
        trigger = {
            livestock = 10
        }
        effect = {
            add_country_modifier = {
                name = "gen_cowboy_modifier" #cav combat and slighly cheaper cav - yeehaw
                duration = 9125
            }
        }
	}
}

#Great Plains and Midwest shared Regional
gen_breadbasket_slot = {
	slot = 5
	generic = yes
	ai = yes

	potential = {
		capital_scope = {
			OR = {
				region = american_midwest_region
				region = great_plains_region
			}
		}
	}
	has_country_shield = no

	gen_breadbasket = { #have lots of grain
		icon = mission_mal_subjugate_zaf
		required_missions = { }
		position = 2
		trigger = {
			OR = {
				trading_bonus = {
					trade_goods = grain
					value = yes
				}
				num_of_owned_provinces_with = {
					value = 20
					trade_goods = grain
					has_manufactory_trigger = yes
				}
			}
		}
		effect = {
			add_country_modifier = {
				name = "bread_basket_of_america_modifier"
				duration = 9125
			}
			custom_tooltip = gen_breadbasket_tooltip
			hidden_effect = {
				every_owned_province = {
					if = {
						limit = {
							trade_goods = grain
							has_manufactory_trigger = yes
						}
						add_base_production = 1
					}
				}
			}
		}
	}
}
